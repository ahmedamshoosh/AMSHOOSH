سـفـيــر الــخــيــر
من نحن أهدافنا خدماتنا الرئيسة

شــعـارنـا يـد تـعـطـي ويـد تـبـنـي

خــدمــاتــنــا -:

  1. مـسـاعـة ذوي الاحـتـجـاجـات الـخـاصـة و الـوقـوـف مـعـهـم .
  2. وتـوفـيـر الـمـيـاه والامـن الغـذائـي .
  3. مـسـاعـدات نـقـديـة و عـيـنـيـة .
  4. كـفـالـة الاـيـتـام ورعـايـتـهـم .
  5. الـزي الـمـدرسـي .
  6. سـلـة اغـاثـيـة .
  7. كـسـوة الـعـيـد .
  8. افـطـار صـائـم .

أ هـد ا فـنـا -:

  1. تـقـد يـم ا لـعـو ن و ا لـمـسـا عـة ا لـنـقـد يـة و ا لـعـيـنـيـة لـلـمـحـتـا جـيـن بـفـئـا تـهـم ا لـمـخـتـلـفـة
  2. ا لا هـتـا م بـا لا يـتـا م و تـقـديـم ا لـر عـا يـة ا لا ز مـة لـهتم
  3. ا لا هـتـمـا م بـذ و ي ا لا حـتـجـا جـا ت ا لـخـا صـة و مـسـا عـد تـهـم
  4. أ قـا مـة و تـنـفـيـذ مـشـا ر يـع ا لا نـسـا نـيـة و ا لا جـتـمـا عـيـة ا لا ز مـة
  5. تـقـد يـم ا لـعـو ن و ا لـمـسـا عـد ة ا لـطـا ر ئـة لـمـن يـحـتـا جـهـا فـي حـا ل ا لـحـو ا د ث و ا لـكـو ا ر ث
  6. ا لـمـسـا هـمـة فـي خـد مـة ا لـمـجـتـمـع و ا قـا مـة ا لا نـشـطـة و ا لا عـمـا ل ا لا جـتـمـا عـيـة و ا لـمـسـا هـمـة فـيـها
  7. ا قـا مـة ا لـبـر ا مـج ا لـتـأ هـيـلـيـة و ا لـتـد ر يـبـيـة وا لـتـعلـيـمـيـة و ا لـثـقـا فـيـة ا لـتـي يـحـتـا جـهـا أ فـر ا د ا لـمـجـتـمـع و تـجـعـل مـنـهـم ا عـضـا ء عـا مـلـيـن نـا فـعـيـن لأ نـفـسـهـم و مـجـتـمـعـهـم
  8. تـو فـيـر نـفـقـا ت ا لـعـلا ج لـلـمـر ضـى ا لـفـقـر ا ء

ر سـا لـتـنـا -:

نـسـعـى أ ن نـقتد م خـد مـا ت خـيـر يـة و تـطـؤ عـيـة نـو عـيـة و غـيـر تـقـلـيـد يـةا نـطـلا قـا مـن خـطـة ا سـتـر ا تـسـجـيـة تـطـو يـر يـة تـبـحـث عـلـى ا لـتـفـؤ ل و ا لـمـنـا فـسـة عـلـى ا لـر يـا د ة فـي ا لـعـمـل ا لـخـيـر ي فـي شـتـى مـجـا لا تـه ( الاجـتـمـا عـيـة - ا لأ نـسـا نـيـة - ا لــتــعـلــيــمـيـة - ا لـصـحـيـة ) جـا مـعـة بـيـن ا لـتـأ صـيـل و ا لا بـد ا ع فـي ا لـعـمـل , لـتـحـسـيـن و ضـع ا لـمـحـتـا جـيـن مـن د ا ئـر ة ا لا حـتـيـا ج ا لـى د ا ئـر ة ا لا عـتـمـا د عـلـى ا لـذ ا ت .

رؤيتنا -:

ر يـا د ة فـي ا لا عـمـا ل ا لـخـيـر يـة و مـسـا هـمـة فـعـا لـة فـي ا لـتـنـمـيـة ا لـمـجـتـمـعـيـة

قـيـمـنـا -:

ا لـشـفـا فيـة , ا لـمـصـدا قـيـة , ا لـشـر ا كـة , ا لـمـسـؤ و لـيـة , ا لـجـو د و ة.

مـن نـحـن -:

نـحـن مـجـمـو عـة سـفـر ا ا لـخـيـر ا لـشـبـا بـيـة نـسـعـى ا لـى تـحـقـيـق ا لـتـكـا فـل ا لا جـتـمـا عـي بـيـن فـئـا ت ا لـمـجـتـمـع مـن خـلا ل تـنـفـيـذ مـشـا ر يـع خـيـر يـة و ا جـتـمـا عـيـة و تـطـو عـيـة .

- تــنـبـع أ هــمــيــة ا لـعـمـل ا لا نـسـا نـي مــن بـؤ س ا لـعـا لـم ا لـمـعـيـش : ا لــلــمــي ء بـا لـحـر و ب و ا لـمــشــا كـل و ا لـصــر ا عــا ت , و ا لـفـقـر و ا لـحـا جـة , و كـمـا ز ا د ت مـآ سـي ا لـمـجـتـمـعـا ت ا شـتـد ت ا لـحـا جـة , و مـن ا لـضـر و ر ي مـد يـد ا لـعـو ن و ا لـمـسـا عـد ة لـلا خـر يـن .
